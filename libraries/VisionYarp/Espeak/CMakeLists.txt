YARP_PREPARE_DEVICE(Espeak TYPE teo::Espeak INCLUDE Espeak.hpp)

IF (NOT SKIP_Espeak)

SET(CMAKE_MODULE_PATH ${ROBOTICSLAB_VISION_MODULE_PATH} ${CMAKE_MODULE_PATH})
FIND_PACKAGE(YARP REQUIRED)
FIND_PACKAGE(Espeak REQUIRED)

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR} ${ROBOTICSLAB_VISION_INCLUDE_DIRS} ${ESPEAK_INCLUDE_DIRS})

yarp_add_idl(IDL_GEN_FILES speech.thrift)

YARP_ADD_PLUGIN(Espeak Espeak.hpp Espeak.cpp ${IDL_GEN_FILES})
TARGET_LINK_LIBRARIES(Espeak ${YARP_LIBRARIES} ${ESPEAK_LIBRARIES})
install(TARGETS Espeak DESTINATION lib)

# Exporting dependencies for ROBOTICSLAB_VISIONConfig.cmake quite manually for now... 
set(ROBOTICSLAB_VISION_INCLUDE_DIRS ${ROBOTICSLAB_VISION_INCLUDE_DIRS} ${ESPEAK_INCLUDE_DIRS} ${CMAKE_CURRENT_SOURCE_DIR} CACHE INTERNAL "appended libraries")
# set(ROBOTICSLAB_VISION_LIBRARIES ${ROBOTICSLAB_VISION_LIBRARIES} VisionYarp Espeak ${ESPEAK_LIBRARIES} CACHE INTERNAL "appended libraries")


#install(TARGETS Espeak DESTINATION lib)
yarp_install(TARGETS Espeak
     EXPORT YARP
     COMPONENT runtime
     LIBRARY DESTINATION ${YARP_DYNAMIC_PLUGINS_INSTALL_DIR}
     ARCHIVE DESTINATION ${YARP_STATIC_PLUGINS_INSTALL_DIR})

yarp_install(FILES Espeak.ini
     COMPONENT runtime
     DESTINATION ${YARP_PLUGIN_MANIFESTS_INSTALL_DIR})

ENDIF (NOT SKIP_Espeak)
